% ----- Edit-preview markup (report-update diff mode) -----
% \reportadd{X}        -> X rendered in blue (new content)
% \reportdel{X}        -> X rendered in grey + strikethrough (removed content)
% \reportchg{old}{new} -> old in grey strikethrough followed by new in blue
%
% After the user accepts the preview, run
%   python3 ../_shared/scripts/accept_edits.py <report-dir>
% which rewrites the source files: \reportadd -> content, \reportdel -> removed,
% \reportchg{a}{b} -> b.
\definecolor{ReportDiffAdd}{HTML}{1F4E79}
\definecolor{ReportDiffDel}{HTML}{8A8F99}
\newcommand{\reportadd}[1]{{\color{ReportDiffAdd}#1}}
\newcommand{\reportdel}[1]{{\color{ReportDiffDel}\sout{#1}}}
\newcommand{\reportchg}[2]{\reportdel{#1}\,\reportadd{#2}}
